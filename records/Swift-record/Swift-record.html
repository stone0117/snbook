能写padding的UILabel

https://blog.csdn.net/l7022995/article/details/79092750

class SNPaddingLabel: UILabel {

  var edgeInsets: UIEdgeInsets!

  override func drawText(in rect: CGRect) {
    //其中用到的self.edgeInsets是我们自己定义的一个属性，在初始化UILabel实例的时候对其赋值
    super.drawText(in: UIEdgeInsetsInsetRect(rect, self.edgeInsets))
  }

  override var intrinsicContentSize: CGSize {
    get {
      var size = super.intrinsicContentSize;
      size.width += (self.edgeInsets.left + self.edgeInsets.right);
      size.height += (self.edgeInsets.top + self.edgeInsets.bottom);
      return size;
    }
  }

  override func sizeThatFits(_ size: CGSize) -> CGSize {
    let superSizeThatFits = super.sizeThatFits(size)
    let width             = superSizeThatFits.width + edgeInsets.left + edgeInsets.right
    let height            = superSizeThatFits.height + edgeInsets.top + edgeInsets.bottom
    return CGSize(width: width, height: height)
  }

  required init?(coder aDecoder: NSCoder) {
    super.init(coder: aDecoder)
    self.edgeInsets = UIEdgeInsets(top: 0, left: 15, bottom: 0, right: 15)
    preferredMaxLayoutWidth = self.frame.width - (self.edgeInsets.left + self.edgeInsets.right)
  }
}

感叹号声明的变量并没有隐式解包, 什么原因?

https://stackoverflow.com/questions/49609528/why-does-implicitly-unwrapped-optional-not-unwrap-in-dictionary-of-type-string


var b: String? = "hello world"
var c: String! = "hello world"

print(b)
print(c)

// 打印 Optional("hello world")
// 打印 some("hello world")

根据SE-0054规定的规则，IUO仅在要求其未包装类型的环境中被强行打开。在您的情况下，IUO不需要强制解包以便被强制Any（Any可以表示任何值），所以它不是。

在这些问答中更详细地讨论了这种行为：

Swift 3错误的字符串插值与隐式解包的Optionals
在Xcode 8中隐式解包可选赋值
您最终ImplicitlyUnwrappedOptional在字典中获得值的事实是在最新的Swift快照中已删除的遗留行为，将来您将最终得到一个Optional值（因为IUO不再是类型）。

然而，这里要注意的一件重要事情（我肯定会绊倒人们）是IUO 的打印在4.1中发生了变化。

在Swift 4.0.3中，您的示例打印如下：

var aString: String! = "hello"
var params : [String : Any] = [
  "myString" : aString
]
print(params)
// This prints ["myString": hello]
给你的幻觉是IUO在被胁迫时被强行打开Any。然而，这就是如何在Swift 4.0.3中打印IUO - 如果它们有一个值，那么它们将打印为该值，否则它们将打印为nil：

var aString: String! = nil
var params : [String : Any] = [
  "myString" : aString
]
print(params)
// This prints ["myString": nil]
之所以在Swift 4.1中改变，是因为在此提交中删除了ImplicitlyUnwrappedOptional符合性，以便在删除类型本身方面取得进展。所以现在使用Swift的默认打印机制（使用反射）打印值。Custom(Debug)StringConvertibleImplicitlyUnwrappedOptional

因此，在字典中，您将获得IUO的默认值debugDescription，如下所示：

let aString: String! = "hello"
let params : [String : Any] = [
  "myString" : aString
]
print(params)
// This prints ["myString": Swift.ImplicitlyUnwrappedOptional<Swift.String>.some("hello")]
如果你自己打印它，你会得到它的默认值description，如下所示：

let aString: String! = "hello"
print(aString) // some("hello")
这是因为在Swift 4.1中，ImplicitlyUnwrappedOptional类型的实现方式Optional与包含两种情况的枚举相同：

public enum ImplicitlyUnwrappedOptional<Wrapped> : ExpressibleByNilLiteral {
  // The compiler has special knowledge of the existence of
  // `ImplicitlyUnwrappedOptional<Wrapped>`, but always interacts with it using
  // the library intrinsics below.

  /// The absence of a value. Typically written using the nil literal, `nil`.
  case none

  /// The presence of a value, stored as `Wrapped`.
  case some(Wrapped)

  // ...
}
对于具有有效负载值的IUO，Swift的默认反射将因此将其打印为some包含包装值的情况。

但这只是暂时的; IUO类型目前（在Swift 4.1中）已弃用，但它将在Swift 4.2中删除。编译器在很多地方内部使用IUO类型，这需要相当多的工作来删除。因此，在4.2中，您将在字典中显示实际 Optional值，这些值将打印出来Optional("hello")。
